一、基础入门（1-10）：Publisher & Subscriber 核心概念
示例 1：最简单的 Publisher + Subscriber
swift
import Combine



// 1. 创建最简单的发布者（Just发布单个值后完成）
let justPublisher = Just("Hello Combine!")

// 2. 订阅发布者（sink是最基础的订阅器）
let cancellable = justPublisher
    .sink(
        receiveCompletion: { completion in
            print("完成状态：\(completion)")
        },
        receiveValue: { value in
            print("收到值：\(value)")
        }
    )

// 输出：
// 收到值：Hello Combine!
// 完成状态：finished
核心说明：Just是最简单的 Publisher，仅发布一个值；sink是最基础的 Subscriber，接收值和完成事件；AnyCancellable用于管理订阅生命周期（防止订阅被提前释放）。


示例 2：使用 CurrentValueSubject（有初始值的可变发布者）
swift
import Combine

// CurrentValueSubject有初始值，修改value会发送新值
let counter = CurrentValueSubject<Int, Never>(0)

let cancellable = counter
    .sink { value in
        print("当前计数：\(value)")
    }

// 修改值触发发布
counter.send(1)
counter.send(2)
counter.value = 3 // 等价于send(3)

// 输出：
// 当前计数：0
// 当前计数：1
// 当前计数：2
// 当前计数：3


示例 3：使用 PassthroughSubject（无初始值的可变发布者）
swift
import Combine

// PassthroughSubject无初始值，仅通过send发送值
let messageSubject = PassthroughSubject<String, Never>()

let cancellable = messageSubject
    .sink(
        receiveCompletion: { _ in print("消息发送完成") },
        receiveValue: { print("收到消息：\($0)") }
    )

messageSubject.send("Hello")
messageSubject.send("World")
messageSubject.send(completion: .finished)

// 输出：
// 收到消息：Hello
// 收到消息：World
// 消息发送完成


示例 4：处理错误（Failure 类型）
swift
import Combine

// 定义自定义错误类型
enum NetworkError: Error {
    case invalidURL
    case requestFailed
}

// PassthroughSubject指定Failure类型为NetworkError
let networkSubject = PassthroughSubject<Data, NetworkError>()

let cancellable = networkSubject
    .sink(
        receiveCompletion: { completion in
            switch completion {
            case .finished:
                print("请求完成")
            case .failure(let error):
                print("请求失败：\(error)")
            }
        },
        receiveValue: { data in
            print("收到数据：\(data.count)字节")
        }
    )

networkSubject.send(Data([1,2,3]))
networkSubject.send(completion: .failure(.requestFailed))


// 输出：
// 收到数据：3字节
// 请求失败：requestFailed
示例 5：AnyCancellable 管理订阅（防止内存泄漏）
swift
import Combine

class ViewModel {
    // 用Set存储所有订阅，自动管理生命周期
    private var cancellables = Set<AnyCancellable>()
    
    func setupPublisher() {
        let publisher = Timer.publish(every: 1, on: .main, in: .common)
            .autoconnect() // 自动连接Timer
        
        publisher
            .sink { [weak self] time in
                print("当前时间：\(time)")
                // 模拟订阅3秒后取消
                if self?.cancellables.count ?? 0 > 0, time.timeIntervalSince1970 > (Date().timeIntervalSince1970 + 3) {
                    self?.cancellables.removeAll()
                }
            }
            .store(in: &cancellables) // 将订阅存入Set
    }
}

let vm = ViewModel()
vm.setupPublisher()
// 输出：每秒打印当前时间，3秒后停止

示例 7：Future 发布者（异步单次操作）
swift
import Combine

// Future用于封装异步操作（如网络请求、异步计算）
func fetchData() -> Future<String, NetworkError> {
    return Future { promise in
        // 模拟异步操作
        DispatchQueue.global().asyncAfter(deadline: .now() + 1) {
            // 成功：promise(.success("数据内容"))
            // 失败：promise(.failure(.requestFailed))
            promise(.success("异步获取的数据"))
        }
    }
}

fetchData()
    .sink(
        receiveCompletion: { print("请求完成：\($0)") },
        receiveValue: { print("获取到数据：\($0)") }
    )
    .store(in: &cancellables)

// 输出（1秒后）：
// 获取到数据：异步获取的数据
// 请求完成：finished

示例 43：网络请求封装（URLSession + Combine）

import Combine
import Foundation

// 封装网络请求（Combine版）
class NetworkManager {
    static let shared = NetworkManager()
    private var cancellables = Set<AnyCancellable>()
    
    func fetch<T: Decodable>(url: URL, type: T.Type) -> AnyPublisher<T, NetworkError> {
        URLSession.shared.dataTaskPublisher(for: url)
            .mapError { error -> NetworkError in
                switch error {
                case .badURL: return .invalidURL
                default: return .requestFailed
                }
            }
            .tryMap { data, response in
                guard let httpResponse = response as? HTTPURLResponse,
                      200..<300 ~= httpResponse.statusCode else {
                    throw NetworkError.requestFailed
                }
                return data
            }
            .decode(type: T.self, decoder: JSONDecoder())
            .mapError { _ in .requestFailed }
            .eraseToAnyPublisher()
    }
}

    let id: Int
    let name: String
}

示例 47：图片缓存（结合 Combine）

import Combine
import UIKit

// 简单的图片缓存
class ImageCache {
    private var cache = NSCache<NSString, UIImage>()
    private var cancellables = Set<AnyCancellable>()
    
    // 下载图片并缓存
    func loadImage(url: URL) -> AnyPublisher<UIImage, Error> {
        // 先查缓存
        if let cachedImage = cache.object(forKey: url.absoluteString as NSString) {
            return Just(cachedImage)
                .setFailureType(to: Error.self)
                .eraseToAnyPublisher()
        }
        
        // 缓存未命中，下载
        return URLSession.shared.dataTaskPublisher(for: url)
            .map { UIImage(data: $0.data)! }
            .receive(on: DispatchQueue.main)
            .handleEvents(receiveOutput: { [weak self] image in
                self?.cache.setObject(image, forKey: url.absoluteString as NSString)
            })
            .eraseToAnyPublisher()
    }
}

示例 48：状态管理（Redux-like）
swift
import Combine

// 简易Redux状态管理
struct AppState {
    var count: Int = 0
}

enum AppAction {
    case increment
    case decrement
}

// Reducer：根据Action更新State
func reducer(state: inout AppState, action: AppAction) {
    switch action {
    case .increment:
        state.count += 1
    case .decrement:
        state.count -= 1
    }
}

// Store：管理状态和动作
class Store: ObservableObject {
    @Published private(set) var state: AppState
    private let actionSubject = PassthroughSubject<AppAction, Never>()
    private var cancellables = Set<AnyCancellable>()
    
    init(initialState: AppState) {
        self.state = initialState
        
        // 监听Action，更新State
        actionSubject
            .scan(initialState) { currentState, action in
                var newState = currentState
                reducer(state: &newState, action: action)
                return newState
            }
            .assign(to: \.state, on: self)
            .store(in: &cancellables)
    }
    
    // 发送Action
    func dispatch(action: AppAction) {
        actionSubject.send(action)
    }
}

// 使用示例
let store = Store(initialState: AppState())
store.$state
    .map { "当前计数：\($0.count)" }
    .sink { print($0) }
    .store(in: &cancellables)

store.dispatch(action: .increment)
store.dispatch(action: .increment)
store.dispatch(action: .decrement)

// 输出：
// 当前计数：0
// 当前计数：1
// 当前计数：2
// 当前计数：1
