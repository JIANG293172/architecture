// MARK: - 1. 基础 Observable 创建
// 作用：学习 just/of/empty 三种最基础序列创建
// 场景：快速生成测试数据、固定值数据流
class Example1_RXSwift_Basic_Observable {
    let disposeBag = DisposeBag()
    
    func test() {
        // just：创建单个元素序列
        Observable<Int>.just(1)
            .subscribe(onNext: { print("just: \($0)") },
                       onCompleted: { print("完成") })
            .disposed(by: disposeBag)
        
        // of：创建多个元素序列
        Observable.of(1,2,3)
            .subscribe(onNext: { print("of: \($0)") })
            .disposed(by: disposeBag)
        
        // empty：空序列，只发完成
        Observable<Void>.empty()
            .subscribe(onCompleted: { print("empty 完成") })
            .disposed(by: disposeBag)
    }
}

// MARK: - 2. 三种 Subject 使用
// 作用：既能发送又能接收事件，是事件中转站
// 场景：页面传值、状态同步、事件通知
class Example2_RXSwift_Basic_Subject {
    let disposeBag = DisposeBag()
    
    func test() {
        // Publish：只收订阅后事件
        let publish = PublishSubject<String>()
        publish.onNext("A")
        publish.subscribe(onNext: { print("Publish: \($0)") })
        publish.onNext("B")
        
        // Behavior：自带最新值
        let behavior = BehaviorSubject(value: "初始值")
        behavior.onNext("C")
        behavior.subscribe(onNext: { print("Behavior: \($0)") })
        behavior.onNext("D")
        
        // Replay：缓存N个历史事件
        let replay = ReplaySubject<Int>.create(bufferSize: 2)
        replay.onNext(1)
        replay.onNext(2)
        replay.onNext(3)
        replay.subscribe(onNext: { print("Replay: \($0)") })
    }
}

// MARK: - 3. 自定义观察者 Observer

    let disposeBag = DisposeBag()
    
    func test() {
        let observer = AnyObserver<String> { event in
            switch event {
            case .next(let v): print("自定义观察: \(v)")

    }
}

    let disposeBag = DisposeBag()
    
    func test() {
        Observable<Int>.create { obs in
            obs.onNext(1)
            obs.onCompleted()
            return Disposables.create()
        }
        .subscribe(on: ConcurrentDispatchQueueScheduler(qos: .background))

    }
}

// MARK: - 19. 失败重试 retry
// 作用：出错自动重试
// 场景：网络请求失败重试
class Example19_RXSwift_Operator_Retry {
    let disposeBag = DisposeBag()
    var count = 0
    
    func test() {
        Observable<Int>.create { [weak self] obs in
            self?.count += 1
            if self!.count < 3 { obs.onError(NSError(domain: "", code: 0)) }
            else { obs.onNext(100); obs.onCompleted() }
            return Disposables.create()
        }
        .retry(2)
        .subscribe(onNext: { print("成功: \($0)") })
        .disposed(by: disposeBag)
    }
}

// MARK: - 29. 网络请求封装
// 作用：Rx + Alamofire 网络请求
// 场景：所有接口请求
class Example29_RXSwift_Network_Request {
    let disposeBag = DisposeBag()
    
    func request(url: String) -> Observable<Data> {
        Observable.create { obs in
            AF.request(url).responseData { res in
                switch res.result {
                case .success(let d): obs.onNext(d); obs.onCompleted()
                case .failure(let e): obs.onError(e)
                }
            }
            return Disposables.create()
        }
    }
}

    let disposeBag = DisposeBag()

    func save(name: String) -> Observable<Bool> {
        Observable.create { obs in
            let ctx = container.viewContext
            let ent = NSEntityDescription.insertNewObject(forEntityName: "User", into: ctx)
            ent.setValue(name, forKey: "name")
            try? ctx.save()
            obs.onNext(true)
            obs.onCompleted()
            return Disposables.create()
        }
    }
}

// MARK: - 43. 条件重试 retryWhen
// 作用：自定义错误重试逻辑
class Example43_RXSwift_Error_Handling_RetryWhen {
    let disposeBag = DisposeBag()
    
    func test() {
        Observable.error(NSError(domain: "", code: 0))
            .retryWhen { err in
                err.flatMapLatest { _ in
                    Observable<Int>.timer(.seconds(1), scheduler: MainScheduler.instance)
                }
            }

// MARK: - 50. 自定义 Observable
// 作用：把任意系统事件包装成 Rx
class Example50_RXSwift_Advanced_Custom_Observable {
    let disposeBag = DisposeBag()
    
    func keyboard() -> Observable<CGFloat> {
        Observable.create { obs in
            let n = NotificationCenter.default.addObserver(
                forName: UIResponder.keyboardWillChangeFrameNotification,
                object: nil,
                queue: .main
            ) { noti in
                let h = (noti.userInfo?[UIResponder.keyboardFrameEndUserInfoKey] as? CGRect)?.height ?? 0
                obs.onNext(h)
            }
            return Disposables.create { NotificationCenter.default.removeObserver(n) }
        }
    }
}
