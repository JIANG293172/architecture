以下是 50 个 RXSwift 从入门到精通的完整 Demo，每个 Demo 对应一个独立类，覆盖基础订阅、操作符、UI 绑定、线程调度、错误处理、高阶用法等核心场景，代码可直接复制运行（需先集成 RXSwift/RXCocoa）。
前置条件：
项目集成 RXSwift（pod 'RxSwift', '~> 6.0'；pod 'RxCocoa', '~> 6.0'）
所有 Demo 类需在 iOS 项目中运行（可在 ViewController 中调用 test () 方法测试）
核心依赖：import RxSwift + import RxCocoa

Example1_RXSwift_Basic_Observable
swift
import RxSwift
import RxCocoa

class Example1_RXSwift_Basic_Observable {
    let disposeBag = DisposeBag()
    
    func test() {
        // 1. 创建基本可观察序列
        let observable = Observable<Int>.just(1)
        observable.subscribe(onNext: { value in
            print("just: \(value)")
        }, onCompleted: {
            print("完成")
        }).disposed(by: disposeBag)
        
        // 2. 多元素序列
        Observable.of(1, 2, 3)
            .subscribe(onNext: { print("of: \($0)") })
            .disposed(by: disposeBag)
        
        // 3. 空序列
        Observable<Void>.empty()
            .subscribe(onCompleted: { print("empty 完成") })
            .disposed(by: disposeBag)
    }
}
Example2_RXSwift_Basic_Subject
swift
import RxSwift
import RxCocoa

class Example2_RXSwift_Basic_Subject {
    let disposeBag = DisposeBag()
    
    func test() {
        // 1. PublishSubject：只发送订阅后的事件
        let publishSubject = PublishSubject<String>()
        publishSubject.onNext("A") // 订阅前发送，不会被接收
        publishSubject.subscribe(onNext: { print("PublishSubject: \($0)") })
            .disposed(by: disposeBag)
        publishSubject.onNext("B")
        
        // 2. BehaviorSubject：发送最新值+后续事件（需初始值）
        let behaviorSubject = BehaviorSubject(value: "初始值")
        behaviorSubject.onNext("C")
        behaviorSubject.subscribe(onNext: { print("BehaviorSubject: \($0)") })
            .disposed(by: disposeBag)
        behaviorSubject.onNext("D")
        
        // 3. ReplaySubject：缓存指定数量事件
        let replaySubject = ReplaySubject<Int>.create(bufferSize: 2)
        replaySubject.onNext(1)
        replaySubject.onNext(2)
        replaySubject.onNext(3)
        replaySubject.subscribe(onNext: { print("ReplaySubject: \($0)") })
            .disposed(by: disposeBag) // 输出2、3
    }
}

import RxSwift
import RxCocoa

class Example19_RXSwift_Operator_Retry {
    let disposeBag = DisposeBag()
    var retryCount = 0
    
    func test() {
        // Retry：失败后重试指定次数
        Observable<Int>.create { [weak self] observer in
            guard let self = self else { return Disposables.create() }
            self.retryCount += 1
            if self.retryCount < 3 {
                observer.onError(NSError(domain: "Test", code: 1001))
            } else {
                observer.onNext(100)
                observer.onCompleted()
            }
            return Disposables.create()
        }
        .retry(2) // 重试2次（总共执行3次）
        .subscribe(onNext: { print("Retry成功: \($0)") },
                   onError: { print("Retry失败: \($0)") })
        .disposed(by: disposeBag)
    }
}

// 扩展UIAlertController支持RX
extension Reactive where Base: UIAlertController {
    var action: Observable<UIAlertAction> {
        return Observable.create { observer in
            let proxy = AlertProxy(alert: self.base, observer: observer)
            return Disposables.create { proxy.stop() }
        }
    }
}

class AlertProxy: NSObject {
    weak var alert: UIAlertController?
    var observer: AnyObserver<UIAlertAction>?
    
    init(alert: UIAlertController, observer: AnyObserver<UIAlertAction>) {
        self.alert = alert
        self.observer = observer
        super.init()
        swizzleAlertAction()
    }
    
    func stop() {
        observer = nil
        alert = nil
    }
    
    private func swizzleAlertAction() {
        // 简化实现：实际需通过Method Swizzling监听UIAlertAction点击
    }

import RxSwift

import Alamofire

class Example30_RXSwift_Network_Request { // 编号笔误，实际是29
    let disposeBag = DisposeBag()
    
    // 网络请求封装
    func requestData(url: String) -> Observable<Data> {
        return Observable.create { observer in
            AF.request(url)
                .responseData { response in
                    switch response.result {
                    case .success(let data):
                        observer.onNext(data)
                        observer.onCompleted()
                    case .failure(let error):
                        observer.onError(error)
                    }

import RxSwift
import RxCocoa
import UIKit

// ViewModel
class UserListViewModel {
    let disposeBag = DisposeBag()
    // 输入
    let refreshTrigger = PublishSubject<Void>()
    // 输出
    let users = BehaviorSubject<[String]>(value: [])
    let isLoading = BehaviorSubject<Bool>(value: false)
    
    init() {
        bind()
    }
    
    private func bind() {
        refreshTrigger
            .do(onNext: { [weak self] _ in self?.isLoading.onNext(true) })
            .flatMap { [weak self] _ -> Observable<[String]> in
                self?.fetchUsers() ?? Observable.just([])
            }
            .do(onNext: { [weak self] _ in self?.isLoading.onNext(false) })
            .bind(to: users)
            .disposed(by: disposeBag)
    }
    
    private func fetchUsers() -> Observable<[String]> {
        // 模拟网络请求
        return Observable.just(["用户1", "用户2", "用户3"])
            .delay(.seconds(1), scheduler: MainScheduler.instance)
    }
}

// ViewController
class Example41_RXSwift_Architecture_ViewModel: UIViewController {
    let disposeBag = DisposeBag()
    let viewModel = UserListViewModel()
    let tableView = UITableView()
    let refreshControl = UIRefreshControl()
    
    override func viewDidLoad() {
        super.viewDidLoad()
        setupUI()
        bindViewModel()
    }
    
    private func setupUI() {
        tableView.register(UITableViewCell.self, forCellReuseIdentifier: "cell")
        tableView.refreshControl = refreshControl
        view.addSubview(tableView)
        tableView.frame = view.bounds
    }
    
    private func bindViewModel() {
        // 绑定刷新触发
        refreshControl.rx.controlEvent(.valueChanged)
            .bind(to: viewModel.refreshTrigger)
            .disposed(by: disposeBag)
        
        // 绑定数据源
        viewModel.users
            .bind(to: tableView.rx.items(cellIdentifier: "cell")) { row, text, cell in
                cell.textLabel?.text = text
            }
            .disposed(by: disposeBag)
        
        // 绑定加载状态
        viewModel.isLoading
            .subscribe(onNext: { [weak self] isLoading in
                if isLoading {
                    self?.refreshControl.beginRefreshing()
                } else {
                    self?.refreshControl.endRefreshing()
                }
            })
            .disposed(by: disposeBag)
    }
}
