# 专利19：一种图片拖拽分享方法

## 技术领域
本发明涉及iOS交互技术领域，特别涉及一种通过拖拽手势快速分享图片的方法。

## 背景技术
分享图片需要：选择图片 → 点击分享 → 选择分享渠道 → 发送，步骤较多。

## 发明内容

### 核心实现
```swift
class DragShareHandler: NSObject, UIDragInteractionDelegate {

    private var sourceView: UIImageView!
    private var shareTargets: [ShareTargetView] = []

    func setup(on imageView: UIImageView) {
        sourceView = imageView

        let dragInteraction = UIDragInteraction(delegate: self)
        imageView.addInteraction(dragInteraction)
        imageView.isUserInteractionEnabled = true
    }

    // MARK: - UIDragInteractionDelegate

    func dragInteraction(_ interaction: UIDragInteraction,
                        itemsForBeginning session: UIDragSession) -> [UIDragItem] {

        guard let image = sourceView.image else { return [] }

        let itemProvider = NSItemProvider(object: image)
        let dragItem = UIDragItem(itemProvider: itemProvider)
        dragItem.localObject = image

        return [dragItem]
    }

    func dragInteraction(_ interaction: UIDragInteraction,
                        previewForLifting item: UIDragItem,
                        session: UIDragSession) -> UITargetedDragPreview? {

        // 自定义拖拽预览
        let previewView = UIImageView(image: item.localObject as? UIImage)
        previewView.frame = CGRect(x: 0, y: 0, width: 100, height: 100)
        previewView.layer.cornerRadius = 8
        previewView.clipsToBounds = true

        let parameters = UIDragPreviewParameters()
        parameters.backgroundColor = .clear

        return UITargetedDragPreview(view: previewView, parameters: parameters)
    }

    func dragInteraction(_ interaction: UIDragInteraction,
                        sessionDidEnd session: UIDragSession) {

        let location = session.location(in: session.view)

        // 检查是否释放在分享目标上
        for target in shareTargets {
            if target.frame.contains(location) {
                target.handleDrop(item: session.items.first?.localObject as? UIImage)
                break
            }
        }
    }
}

// 分享目标视图
class ShareTargetView: UIView {

    var onDrop: ((UIImage) -> Void)?

    func handleDrop(item: UIImage?) {
        guard let image = item else { return }

        // 显示分享预览
        UIView.animate(withDuration: 0.2) {
            self.transform = CGAffineTransform(scaleX: 1.2, y: 1.2)
        } completion: { _ in
            self.onDrop?(image)
            UIView.animate(withDuration: 0.2) {
                self.transform = .identity
            }
        }
    }
}
```

## 优点
1. 拖拽分享直观自然
2. 自定义预览样式
3. 可拖拽到指定分享目标
