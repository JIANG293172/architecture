# 专利12：一种实况锁屏动态车辆壁纸的方法

## 技术领域
本发明涉及iOS壁纸技术领域，特别涉及一种根据车辆状态动态变化的锁屏壁纸。

## 背景技术
用户每天多次看到锁屏壁纸，但现有壁纸静态不变，无法反映车辆实时状态。

## 发明内容

### 核心实现
```swift
// Live Activity 动态壁纸
@available(iOS 16.1, *)
class VehicleLiveActivity: Widget {

    var body: some WidgetConfiguration {
        ActivityConfiguration(for: VehicleStatus.self) { context in
            // 锁屏小组件样式
            HStack {
                Image(systemName: vehicleIcon(for: context.state.status))
                    .font(.largeTitle)

                VStack(alignment: .leading) {
                    Text(context.state.vehicleName)
                        .font(.headline)
                    Text(context.state.statusText)
                        .font(.caption)
                        .foregroundColor(.secondary)
                }

                Spacer()

                // 电量可视化
                BatteryView(level: context.state.batteryLevel)
            }
            .padding()
        } dynamicIsland: { context in
            DynamicIsland {
                // 展开样式
                DynamicIslandExpandedRegion(.leading) {
                    Image(systemName: vehicleIcon(for: context.state.status))
                }
                DynamicIslandExpandedRegion(.trailing) {
                    Text("\(context.state.batteryLevel)%")
                }
            } compactLeading: {
                Image(systemName: "car.fill")
            } compactTrailing: {
                Text("\(context.state.batteryLevel)%")
            }
        }
    }

    private func vehicleIcon(for status: VehicleStatus) -> String {
        switch status {
        case .locked: return "lock.fill"
        case .unlocked: return "lock.open.fill"
        case .running: return "car.fill"
        case .charging: return "bolt.car.fill"
        }
    }
}

// 更新实况活动
class VehicleStatusManager {

    func updateLiveActivity() {
        let state = VehicleAttributes.ContentState(
            status: .locked,
            vehicleName: "长安 UNI-K",
            batteryLevel: 75,
            statusText: "已锁车 · 剩余75%"
        )

        // 更新或创建实况活动
        Task {
            for await activity in Activity<VehicleStatus>.activities {
                await activity.update(using: state)
            }
        }
    }
}
```

## 优点
1. 锁屏直观显示车辆状态
2. 电量、车门状态一目了然
3. 品牌感强，用户粘性高
