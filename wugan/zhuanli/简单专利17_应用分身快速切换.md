# 专利17：一种应用分身快速切换方法

## 技术领域
本发明涉及iOS应用切换技术领域，特别涉及一种多账号快速切换的交互方法。

## 背景技术
车控APP支持家庭多成员共享一辆车，用户需要在不同账号间切换，现有切换方式需要注销再登录，步骤繁琐。

## 发明内容

### 核心实现
```swift
class AccountSwitchManager {

    static let shared = AccountSwitchManager()

    private var cachedAccounts: [CachedAccount] = []
    private let maxCachedAccounts = 3

    func cacheCurrentAccount() {
        let current = getCurrentAccount()
        guard let account = current else { return }

        // 添加到缓存
        if !cachedAccounts.contains(where: { $0.id == account.id }) {
            cachedAccounts.insert(account, at: 0)

            // 限制缓存数量
            if cachedAccounts.count > maxCachedAccounts {
                cachedAccounts.removeLast()
            }
        }

        saveToUserDefaults()
    }

    func switchToAccount(_ account: CachedAccount) {
        // 保存当前账号
        cacheCurrentAccount()

        // 切换账号
        performAccountSwitch(to: account)
    }

    func showSwitcher(from viewController: UIViewController) {
        let sheet = UIAlertController(title: "切换账号", message: nil, preferredStyle: .actionSheet)

        for account in cachedAccounts {
            let action = UIAlertAction(title: account.displayName, style: .default) { _ in
                self.switchToAccount(account)
            }

            // 显示当前账号
            if account.id == getCurrentAccount()?.id {
                action.setValue(true, forKey: "checked")
            }

            sheet.addAction(action)
        }

        sheet.addAction(UIAlertAction(title: "+ 添加账号", style: .default) { _ in
            // 添加新账号
        })

        sheet.addAction(UIAlertAction(title: "取消", style: .cancel))

        viewController.present(sheet, animated: true)
    }
}

// 快捷切换入口：摇一摇
extension UIWindow {

    override func motionEnded(_ motion: UIEvent.EventSubtype, with event: UIEvent?) {
        if motion == .motionShake {
            if let vc = rootViewController as? MainTabBarController {
                AccountSwitchManager.shared.showSwitcher(from: vc)
            }
        }
    }
}
```

## 优点
1. 缓存最近使用账号，一键切换
2. 摇一摇呼出切换器
3. 无需注销登录
